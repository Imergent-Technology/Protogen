```php
// Enhanced authentication middleware
class RoleBasedAuthMiddleware
{
    public function handle($request, Closure $next, $role)
    {
        $user = $request->user();
        
        if (!$user || !$this->hasRole($user, $role)) {
            return response()->json(['error' => 'Unauthorized'], 401);
        }
        
        return $next($request);
    }
}
```
